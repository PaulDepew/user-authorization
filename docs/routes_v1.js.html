router.get('/:model', handleGetUsers);
router.post('/:model', handleCreateUser);
router.post('/:model', authMiddleware, handleSignIn);

/**
 * HandleGetUser - gets all users
 * @function handleGetUsers
 * @param {*} request 
 * @param {*} response 
 * @returns {object}
 */
async function handleGetUsers(req, res){
  res.send(new User(req));
}


/**
 * handleCreateUser - creates a user based on the body
 * @function handleCreateUser
 * @param {*} request 
 * @param {*} response 
 * @returns {object}
 */
async function handleCreateUser(req, res){
  res.send(new User(req.body).generateToken());
}


/**
 * handleSignIn - authorizes and signs in a user
 * @function handleSignIn
 * @param {*} request 
 * @param {*} response 
 * @returns {object}
 */
async function handleSignIn(req, res){
  res.send('Working on signing in');
}
