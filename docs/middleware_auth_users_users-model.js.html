'use strict';

/**
 * Creates a Schema and hashed password for the User
 * @module User
 * @param UserSchema
 * @returns {object}
 */

const user = require('./users-schema.js');
const bcrypt = require('bcrypt');
const base64 = require('base-64');

class User {
  constructor(user){
    this.username = user.username;
    this.password = this.hashPass(user.password);
  }

  async hashPass(value){
    return await bcrypt.hash(value, 10);
  }
  
  generateToken(){
    let payload = `${this.username}:${this.password}`;
    let encodedPayload = base64.encode(payload);
    return encodedPayload;
  }
}

module.exports = User;
